clean_txt<-function(txt){
  txt<-tolower(txt)             # ,  
  txt<-removePunctuation(txt)   #  
  txt<-removeNumbers(txt)       #  
  txt<-stripWhitespace(txt)     # 
  
  return(txt)
}
